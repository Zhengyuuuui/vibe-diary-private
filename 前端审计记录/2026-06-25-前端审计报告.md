### P0-1: DiaryReaderView 翻页引擎内存泄漏风险

**📍 位置**:  
[client/src/views/DiaryReaderView.vue#L626-L646](file:///Users/zhengyusheng/Desktop/diary备份/client/src/views/DiaryReaderView.vue#L626-L646)

**🔍 问题描述**:  
`onBeforeUnmount` 中虽然调用了 `destroy()`,但 `flipContainer.value.innerHTML = ''` 可能无法完全清除 StPageFlip 创建的复杂 DOM 结构(包含 `stf__parent`, `stf__wrapper`, `stf__block` 等多层包裹)。

**⚠️ 影响范围**:  
- 频繁进出日记页面可能导致内存累积
- 移动端设备内存敏感,更容易触发崩溃
- 长时间使用后浏览器性能下降

**🆕 建议方案**:  
```javascript
// 🆕 改进的清理逻辑
onBeforeUnmount(() => {
  isUnmounted.value = true
  
  if (pageFlipInstance.value) {
    try {
      pageFlipInstance.value.destroy()
    } catch (e) {}
    pageFlipInstance.value = null
  }
  
  // 🆕 深度清理 StPageFlip 生成的节点
  if (flipContainer.value) {
    const stfNodes = flipContainer.value.querySelectorAll('[class*="stf__"]')
    stfNodes.forEach(node => {
      try { node.remove() } catch (_) {}
    })
    flipContainer.value.innerHTML = ''
  }
  
  // 🆕 清除所有自动保存定时器
  Object.values(saveTimers.value).forEach(clearTimeout)
  saveTimers.value = {}
})
```

**✅ 验证方法**:  
1. Chrome DevTools Memory Profiler
2. 进出日记页面10次,观察内存曲线
3. 检查 DOM 树残留节点

### P0-3: Token存储安全风险

**📍 位置**:  
[client/src/stores/auth.js#L6-L16](file:///Users/zhengyusheng/Desktop/diary备份/client/src/stores/auth.js#L6-L16)

**🔍 问题描述**:  
Token 直接存储在 localStorage,无加密保护。XSS 攻击可窃取 token,导致用户隐私数据泄露。

**⚠️ 影响范围**:  
- 用户登录态可被窃取
- 所有日记内容、收藏内容可被访问
- 符合 GDPR/隐私法规风险

**🆕 建议方案**:  
```javascript
// 🆕 方案1: 使用 sessionStorage (关闭浏览器即失效)
const token = ref(sessionStorage.getItem('token') || '')

// 🆕 方案2: 加密存储
import CryptoJS from 'crypto-js'

const SECRET_KEY = 'your-secret-key' // 🆕 应从环境变量读取

function encryptToken(token) {
  return CryptoJS.AES.encrypt(token, SECRET_KEY).toString()
}

function decryptToken(encrypted) {
  const bytes = CryptoJS.AES.decrypt(encrypted, SECRET_KEY)
  return bytes.toString(CryptoJS.enc.Utf8)
}

// 🆕 存储时加密
localStorage.setItem('token', encryptToken(data.token))

// 🆕 读取时解密
const encrypted = localStorage.getItem('token')
token.value = encrypted ? decryptToken(encrypted) : ''
```

**✅ 验证方法**:  
1. 检查 localStorage 是否有明文 token
2. XSS 攻击模拟测试
3. 安全审计工具扫描

### P1-3: API错误处理不统一

**📍 位置**:  
[client/src/api/index.js#L7-L23](file:///Users/zhengyusheng/Desktop/diary备份/client/src/api/index.js#L7-L23)

**🔍 问题描述**:  
所有 API 调用只抛出错误,未统一处理网络异常、超时等情况。用户网络不稳定时可能看到空白页面或无提示。

**⚠️ 影响范围**:  
- 用户体验差
- 错误信息不友好
- 无法自动重试

**🆕 建议方案**:  
```javascript
// 🆕 全局错误拦截器
export async function request(url, options = {}) {
  const token = getToken()
  const headers = { 'Content-Type': 'application/json' }
  if (token) {
    headers['Authorization'] = `Bearer ${token}`
  }
  
  // 🆕 添加超时控制
  const timeout = options.timeout || 10000
  const controller = new AbortController()
  const timeoutId = setTimeout(() => controller.abort(), timeout)
  
  try {
    const res = await fetch(`${BASE}${url}`, {
      headers,
      ...options,
      signal: controller.signal
    })
    clearTimeout(timeoutId)
    
    const json = await res.json()
    
    // 🆕 统一错误处理
    if (json.code !== 200 && json.code !== 201) {
      throw new Error(json.msg || '请求失败')
    }
    
    return json.data
  } catch (err) {
    clearTimeout(timeoutId)
    
    // 🆕 网络错误统一提示
    if (err.name === 'AbortError') {
      showToast('请求超时,请检查网络连接')
    } else if (!navigator.onLine) {
      showToast('网络已断开,请检查网络设置')
    } else {
      showToast(err.message || '网络请求失败')
    }
    
    throw err
  }
}

// 🆕 Toast 提示函数
function showToast(message) {
  // 使用现有的 Toast 组件
  const toast = document.createElement('div')
  toast.className = 'fixed bottom-24 left-1/2 transform -translate-x-1/2 z-50 bg-surface-container-high px-6 py-3 rounded-full shadow-lg'
  toast.textContent = message
  document.body.appendChild(toast)
  setTimeout(() => toast.remove(), 3000)
}
```

**📍 位置**:  
[client/src/router.js#L48-L58](file:///Users/zhengyusheng/Desktop/diary备份/client/src/router.js#L48-L58)

**🔍 问题描述**:  
仅检查 token 是否存在,未验证是否过期。过期 token 仍可进入页面,后续 API 调用会失败。

**⚠️ 影响范围**:  
- 用户进入页面后立即报错
- 体验不连贯
- 需要重新登录

**🆕 建议方案**:  
```javascript
// 🆕 改进的路由守卫
router.beforeEach(async (to, from, next) => {
  const token = localStorage.getItem('token')
  
  if (to.meta.requiresAuth) {
    if (!token) {
      next('/login')
      return
    }
    
    // 🆕 验证 token 有效性
    try {
      const authStore = useAuthStore()
      const isValid = await authStore.checkAuth()
      
      if (!isValid) {
        next('/login')
        return
      }
    } catch (error) {
      console.error('Token验证失败:', error)
      next('/login')
      return
    }
  }
  
  if (to.path === '/login' && token) {
    // 🆕 已登录用户访问登录页,重定向到首页
    next('/')
    return
  }
  
  next()
})
```

### P2-8: 性能 - 图片未压缩优化

**🔍 问题描述**:  
上传的图片未在前端压缩,大图片占用带宽和存储空间。

**🆕 建议方案**:  
```javascript
// 🆕 图片压缩工具
async function compressImage(file, maxWidth = 800, quality = 0.8) {
  return new Promise((resolve) => {
    const reader = new FileReader()
    reader.onload = (e) => {
      const img = new Image()
      img.onload = () => {
        const canvas = document.createElement('canvas')
        
        // 🆕 计算压缩尺寸
        let width = img.width
        let height = img.height
        
        if (width > maxWidth) {
          height = (height * maxWidth) / width
          width = maxWidth
        }
        
        canvas.width = width
        canvas.height = height
        
        const ctx = canvas.getContext('2d')
        ctx.drawImage(img, 0, 0, width, height)
        
        // 🆕 压缩为 JPEG
        canvas.toBlob(
          (blob) => resolve(blob),
          'image/jpeg',
          quality
        )
      }
      img.src = e.target.result
    }
    reader.readAsDataURL(file)
  })
}

// 🆕 改进的上传函数
export async function uploadAvatar(formData) {
  const file = formData.get('avatar')
  
  // 🆕 压缩图片
  if (file && file.type.startsWith('image/')) {
    const compressed = await compressImage(file, 400, 0.8)
    formData.set('avatar', compressed, file.name)
  }
  
  const token = getToken()
  return fetch(`${BASE}/auth/avatar`, {
    method: 'POST',
    headers: { 'Authorization': `Bearer ${token}` },
    body: formData
  }).then(res => res.json())
    .then(json => {
      if (json.code !== 200) throw new Error(json.msg)
      return json.data
    })
}
```

# 回忆卡片功能开发文档

## 功能概述

在 `BookshelfView.vue` 页面的 `bg-surface-container-low p-8 rounded-xl` 卡片中实现两个功能：

1. **回忆碎片**：随机展示一篇过去日记的片段
2. **历史上的今天**：显示去年/前年同月同日写的内容

## 功能一：回忆碎片

### 功能说明

- 从用户所有日记中随机抽取一篇日记的片段
- 显示日记本标题、日期、内容前100字
- 点击可跳转查看完整内容
- 点击刷新按钮换一篇

### 后端接口设计

#### 新增接口：`GET /api/v1/diaries/random-fragment`

**文件位置**：`/server/routes/diaries.js`

**请求参数**：无

**返回数据**：

```json
{
  "code": 200,
  "msg": "获取成功",
  "data": {
    "diary_id": 5,
    "diary_title": "秋天的故事",
    "page_id": 23,
    "content": "落叶铺满了小路，踩上去发出清脆的声响...",
    "created_at": "2023-10-15T14:30:00",
    "has_more": true
  }
}
```

**后端实现代码**：

```javascript
router.get('/random-fragment', authMiddleware, (req, res) => {
  const db = req.db;
  const userId = req.user.id;

  // 获取用户所有有内容的页面，随机抽取一条
  const page = db.prepare(`
    SELECT p.id as page_id, p.content, p.created_at, p.diary_id, d.title as diary_title
    FROM pages p
    JOIN diaries d ON p.diary_id = d.id
    WHERE d.user_id = ? AND p.content IS NOT NULL AND p.content != ''
    ORDER BY RANDOM()
    LIMIT 1
  `).get(userId);

  if (!page) {
    return res.json({
      code: 200,
      msg: '暂无日记',
      data: null
    });
  }

  // 截取前100个字符作为片段
  const fragment = page.content.substring(0, 100);
  const hasMore = page.content.length > 100;

  res.json({
    code: 200,
    msg: '获取成功',
    data: {
      diary_id: page.diary_id,
      diary_title: page.diary_title,
      page_id: page.page_id,
      content: fragment,
      created_at: page.created_at,
      has_more: hasMore
    }
  });
});
```

## 功能二：历史上的今天

### 功能说明

- 查询用户在去年、前年等同月同日写的日记
- 优先显示最近的年份（去年 > 前年 > 大前年）
- 如果没有记录，显示提示文字
- 点击可跳转查看完整内容

### 后端接口设计

#### 新增接口：`GET /api/v1/diaries/on-this-day`

**文件位置**：`/server/routes/diaries.js`

**请求参数**：无（使用服务器当前日期）

**返回数据**：

```json
{
  "code": 200,
  "msg": "获取成功",
  "data": {
    "diary_id": 3,
    "diary_title": "晴天日记",
    "page_id": 15,
    "content": "今天阳光很好，我坐在窗边看了一下午的书...",
    "created_at": "2024-04-18T16:20:00",
    "year": 2024,
    "has_more": true
  }
}
```

**无数据时返回**：

```json
{
  "code": 200,
  "msg": "暂无记录",
  "data": null
}
```

**后端实现代码**：

```javascript
router.get('/on-this-day', authMiddleware, (req, res) => {
  const db = req.db;
  const userId = req.user.id;

  // 获取今天的月和日
  const today = new Date();
  const month = String(today.getMonth() + 1).padStart(2, '0');
  const day = String(today.getDate()).padStart(2, '0');
  const monthDay = `-${month}-${day}`;

  // 查询去年、前年等同月同日的日记（最多查3年前）
  const currentYear = today.getFullYear();
  const years = [currentYear - 1, currentYear - 2, currentYear - 3];

  for (const year of years) {
    const page = db.prepare(`
      SELECT p.id as page_id, p.content, p.created_at, p.diary_id, d.title as diary_title
      FROM pages p
      JOIN diaries d ON p.diary_id = d.id
      WHERE d.user_id = ? 
      AND p.content IS NOT NULL 
      AND p.content != ''
      AND strftime('%Y-%m-%d', p.created_at) = ?
      ORDER BY p.created_at DESC
      LIMIT 1
    `).get(userId, `${year}${monthDay}`);

    if (page) {
      const fragment = page.content.substring(0, 100);
      const hasMore = page.content.length > 100;

      return res.json({
        code: 200,
        msg: '获取成功',
        data: {
          diary_id: page.diary_id,
          diary_title: page.diary_title,
          page_id: page.page_id,
          content: fragment,
          created_at: page.created_at,
          year: year,
          has_more: hasMore
        }
      });
    }
  }

  // 没有找到历史记录
  res.json({
    code: 200,
    msg: '暂无记录',
    data: null
  });
});
```

## 前端实现

### 1. 添加 API 调用

**文件位置**：`/client/src/api/index.js`

```javascript
export function getRandomFragment() {
  return request('/diaries/random-fragment')
}

export function getOnThisDay() {
  return request('/diaries/on-this-day')
}
```

### 2. 修改 BookshelfView.vue

**文件位置**：`/client/src/views/BookshelfView.vue`

```vue
<script setup>
import { getRandomFragment, getOnThisDay } from '@/api'

// 回忆碎片数据
const memoryFragment = ref(null)

// 历史上的今天数据
const onThisDay = ref(null)

// 加载回忆碎片
async function loadMemoryFragment() {
  try {
    const data = await getRandomFragment()
    memoryFragment.value = data
  } catch (error) {
    console.error('加载回忆碎片失败:', error)
  }
}

// 加载历史上的今天
async function loadOnThisDay() {
  try {
    const data = await getOnThisDay()
    onThisDay.value = data
  } catch (error) {
    console.error('加载历史上的今天失败:', error)
  }
}

// 刷新回忆碎片
function refreshFragment() {
  loadMemoryFragment()
}

// 格式化日期显示
function formatDate(dateStr) {
  if (!dateStr) return ''
  const date = new Date(dateStr)
  return date.toLocaleDateString('zh-CN', {
    year: 'numeric',
    month: 'long',
    day: 'numeric'
  })
}

// 格式化年份显示
function formatYear(dateStr) {
  if (!dateStr) return ''
  return new Date(dateStr).getFullYear()
}

// 跳转到日记
function goToDiary(diaryId) {
  router.push(`/diary/${diaryId}`)
}

// 在 onMounted 中调用
onMounted(async () => {
  // ... 现有代码
  await loadMemoryFragment()
  await loadOnThisDay()
})
</script>

<template>
<!-- 替换现有的 bg-surface-container-low p-8 rounded-xl 卡片 -->
<div class="bg-surface-container-low p-8 rounded-xl relative overflow-hidden group">
  <!-- 右上角刷新按钮 -->
  <div class="absolute top-0 right-0 p-4 flex items-center gap-2">
    <button 
      @click="refreshFragment" 
      class="p-2 rounded-full hover:bg-surface-container transition-colors"
      title="换一篇回忆"
    >
      <span class="material-symbols-outlined text-outline-variant hover:text-primary transition-colors">refresh</span>
    </button>
  </div>

  <!-- 历史上的今天 -->

    <h4 class="font-label text-xs uppercase tracking-widest text-on-surface-variant mb-3 flex items-center gap-2">
      <span class="material-symbols-outlined text-sm">history</span>
      历史上的今天
    </h4>
    
    <div v-if="onThisDay" 
      class="bg-surface-container-lowest/50 p-4 rounded-lg cursor-pointer hover:bg-surface-container-lowest/80 transition-colors"
      @click="goToDiary(onThisDay.diary_id)"
    >
      <div class="flex items-center gap-2 mb-2">
        <span class="font-label text-[10px] text-primary">{{ onThisDay.year }}年{{ new Date().toLocaleDateString('zh-CN', { month: 'long', day: 'numeric' }) }}</span>
        <span class="text-on-surface-variant/40">·</span>
        <span class="font-label text-[10px] text-on-surface-variant">{{ onThisDay.diary_title }}</span>
      </div>
      <p class="text-on-surface-variant text-sm leading-relaxed line-clamp-2">"{{ onThisDay.content }}{{ onThisDay.has_more ? '...' : '' }}"</p>
      <span class="font-label text-[10px] text-primary mt-2 inline-block hover:underline">查看完整内容 →</span>
    </div>
    
    <div v-else class="bg-surface-container-lowest/30 p-4 rounded-lg text-center">
      <span class="material-symbols-outlined text-3xl text-outline-variant mb-2">history</span>
      <p class="text-on-surface-variant text-sm">那年今日，你还没有开始书写</p>
      <p class="text-on-surface-variant/60 text-xs mt-1">开始记录，让未来的你拥有回忆</p>
    </div>
  </div>

  <!-- 分隔线 -->
  <div class="border-t border-outline-variant/20 my-6"></div>

  <!-- 回忆碎片 -->
  <div>
    <h4 class="font-label text-xs uppercase tracking-widest text-on-surface-variant mb-3 flex items-center gap-2">
      <span class="material-symbols-outlined text-sm">auto_awesome</span>
      回忆碎片
    </h4>
    
    <div v-if="memoryFragment" 
      class="bg-surface-container-lowest/50 p-4 rounded-lg cursor-pointer hover:bg-surface-container-lowest/80 transition-colors"
      @click="goToDiary(memoryFragment.diary_id)"
    >
      <div class="flex items-center gap-2 mb-2">
        <span class="font-label text-[10px] text-on-surface-variant">{{ memoryFragment.diary_title }}</span>
        <span class="text-on-surface-variant/40">·</span>
        <span class="font-label text-[10px] text-on-surface-variant/60">{{ formatDate(memoryFragment.created_at) }}</span>
      </div>
      <p class="text-on-surface-variant text-sm leading-relaxed line-clamp-2">"{{ memoryFragment.content }}{{ memoryFragment.has_more ? '...' : '' }}"</p>
      <span class="font-label text-[10px] text-primary mt-2 inline-block hover:underline">查看完整内容 →</span>
    </div>
    
    <div v-else class="bg-surface-container-lowest/30 p-4 rounded-lg text-center">
```

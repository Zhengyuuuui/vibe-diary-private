# 灵感收藏功能开发文档

## 一、功能概述

### 1.1 目标
为 FavoritesView 页面添加完整的灵感收藏功能，支持：
- 从日记中提取内容收藏
- 独立创建灵感内容
- 多维度筛选和搜索
- 随机灵感发现
- 统计数据展示
- 灵感板创作（高级功能）

### 1.2 设计原则
- ✅ **不破坏现有功能**：所有新功能通过新增实现，不修改现有表结构
- ✅ **保持视觉风格**：延续现有的优雅设计和动画效果
- ✅ **渐进式开发**：分阶段实现，每个阶段可独立运行
- ✅ **向后兼容**：支持从静态数据平滑迁移到动态数据

## 二、数据库设计

### 2.1 新增表结构

#### `favorites` 表（收藏主表）
```sql
CREATE TABLE IF NOT EXISTS favorites (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id INTEGER NOT NULL,
  
  -- 内容类型
  type TEXT NOT NULL CHECK(type IN ('quote', 'image', 'vertical', 'featured', 'minimal')),
  
  -- 基础字段
  title TEXT,
  content TEXT NOT NULL,
  source TEXT,           -- 引用来源
  image TEXT,            -- 图片 URL
  badge TEXT,            -- 徽章文字（如"年度精选"）
  
  -- 标签和心情（JSON 格式）
  tags TEXT,             -- JSON 数组：["冥想", "夜读"]
  mood TEXT,             -- 心情：calm, joyful, melancholy, inspired, peaceful
  
  -- 关联字段（可选）
  diary_id INTEGER,      -- 关联的日记本 ID（从日记提取时使用）
  page_id INTEGER,       -- 关联的页面 ID（从日记提取时使用）
  
  -- 时间戳
  created_at TEXT NOT NULL DEFAULT (datetime('now', 'localtime')),
  updated_at TEXT NOT NULL DEFAULT (datetime('now', 'localtime')),
  
  -- 外键约束
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (diary_id) REFERENCES diaries(id) ON DELETE SET NULL,
  FOREIGN KEY (page_id) REFERENCES pages(id) ON DELETE SET NULL
);

-- 索引优化
CREATE INDEX IF NOT EXISTS idx_favorites_user_id ON favorites(user_id);
CREATE INDEX IF NOT EXISTS idx_favorites_type ON favorites(type);
CREATE INDEX IF NOT EXISTS idx_favorites_mood ON favorites(mood);
CREATE INDEX IF NOT EXISTS idx_favorites_created_at ON favorites(created_at);
```

const { Router } = require('express');
const { authMiddleware } = require('../middleware/auth');
const router = Router();

// 获取收藏列表（支持筛选、搜索、分页）
router.get('/', authMiddleware, (req, res) => {
  const db = req.db;
  const userId = req.user.id;
  
  // 查询参数
  const page = Math.max(1, parseInt(req.query.page) || 1);
  const limit = Math.min(50, Math.max(1, parseInt(req.query.limit) || 20));
  const offset = (page - 1) * limit;
  
  // 筛选条件
  const type = req.query.type;      // 类型筛选
  const mood = req.query.mood;      // 心情筛选
  const search = req.query.search;  // 搜索关键词
  
  // 构建查询
  let sql = 'SELECT * FROM favorites WHERE user_id = ?';
  const params = [userId];
  
  if (type) {
    sql += ' AND type = ?';
    params.push(type);
  }
  
  if (mood) {
    sql += ' AND mood = ?';
    params.push(mood);
  }
  
  if (search) {
    sql += ' AND (title LIKE ? OR content LIKE ? OR source LIKE ?)';
    const searchPattern = `%${search}%`;
    params.push(searchPattern, searchPattern, searchPattern);
  }
  
  // 统计总数
  const countSql = sql.replace('SELECT *', 'SELECT COUNT(*) as count');
  const total = db.prepare(countSql).get(...params).count;
  
  // 分页查询
  sql += ' ORDER BY created_at DESC LIMIT ? OFFSET ?';
  params.push(limit, offset);
  
  const favorites = db.prepare(sql).all(...params);
  
  // 解析 JSON 字段
  const parsedFavorites = favorites.map(fav => ({
    ...fav,
    tags: fav.tags ? JSON.parse(fav.tags) : []
  }));
  
  res.json({
    code: 200,
    msg: '获取成功',
    data: {
      list: parsedFavorites,
      total,
      page,
      limit
    }
  });
});

router.post('/', authMiddleware, (req, res) => {

  const userId = req.user.id;

    type,
    title,
    content,
    source,
    image,
    badge,
    tags,
    mood,
    diary_id,

  if (!validTypes.includes(type)) {
    return res.status(400).json({
      code: 400,
      msg: '无效的类型',
      data: null

  // 插入数据
  const result = db.prepare(`
    INSERT INTO favorites (
      user_id, type, title, content, source, image, badge, 
      tags, mood, diary_id, page_id

    source || null,
    image || null,

    msg: '创建成功',

router.get('/stats/overview', authMiddleware, (req, res) => {
  const db = req.db;
  const userId = req.user.id;
  
  // 总数
  const total = db.prepare(
    'SELECT COUNT(*) as count FROM favorites WHERE user_id = ?'
  ).get(userId).count;
  
  // 按类型统计
  const byType = db.prepare(`
    SELECT type, COUNT(*) as count 
    FROM favorites 
    WHERE user_id = ? 
    GROUP BY type
  `).all(userId);
  
  // 按心情统计
  const byMood = db.prepare(`
    SELECT mood, COUNT(*) as count 
    FROM favorites 
    WHERE user_id = ? AND mood IS NOT NULL
    GROUP BY mood
  `).all(userId);
  
  // 本月新增
  const thisMonth = db.prepare(`
    SELECT COUNT(*) as count 
    FROM favorites 
    WHERE user_id = ? 
    AND strftime('%Y-%m', created_at) = strftime('%Y-%m', 'now', 'localtime')
  `).get(userId).count;
  
  // 最近使用的标签
  const recentTags = db.prepare(`
    SELECT tags 
    FROM favorites 
    WHERE user_id = ? AND tags IS NOT NULL 
    ORDER BY created_at DESC 
    LIMIT 20
  `).all(userId);
  
  // 统计标签频率
  const tagFrequency = {};
  recentTags.forEach(row => {
    const tags = JSON.parse(row.tags);
    tags.forEach(tag => {
      tagFrequency[tag] = (tagFrequency[tag] || 0) + 1;
    });
  });
  
  const topTags = Object.entries(tagFrequency)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 10)
    .map(([tag, count]) => ({ tag, count }));
  
  res.json({
    code: 200,
    msg: '获取成功',
    data: {
      total,
      thisMonth,
      byType,
      byMood,
      topTags
    }
  });
});

## 四、前端架构设计

### 4.1 Pinia Store：`client/src/stores/favorites.js`

```javascript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { request } from '@/api'

export const useFavoritesStore = defineStore('favorites', () => {
  const favorites = ref([])
  const currentFavorite = ref(null)
  const stats = ref(null)
  const loading = ref(false)
  
  const filters = ref({
    type: null,
    mood: null,
    search: ''
  })
  
  const pagination = ref({
    page: 1,
    limit: 20,
    total: 0
  })
  
  const filteredFavorites = computed(() => {
    let result = favorites.value
    
    if (filters.value.type) {
      result = result.filter(f => f.type === filters.value.type)
    }
    
    if (filters.value.mood) {
      result = result.filter(f => f.mood === filters.value.mood)
    }
    
    if (filters.value.search) {
      const query = filters.value.search.toLowerCase()
      result = result.filter(f => 
        f.title?.toLowerCase().includes(query) ||
        f.content?.toLowerCase().includes(query) ||
        f.source?.toLowerCase().includes(query)
      )
    }
    
    return result
  })
  
  async function loadFavorites(params = {}) {
    loading.value = true
    try {
      const query = new URLSearchParams({
        page: params.page || pagination.value.page,
        limit: params.limit || pagination.value.limit,
        ...(filters.value.type && { type: filters.value.type }),
        ...(filters.value.mood && { mood: filters.value.mood }),
        ...(filters.value.search && { search: filters.value.search })
      })
      
      const data = await request(`/favorites?${query}`)
      favorites.value = data.list
      pagination.value.total = data.total
      pagination.value.page = data.page
    } finally {
      loading.value = false
    }
  }
  
  async function loadFavorite(id) {
    loading.value = true
    try {
      currentFavorite.value = await request(`/favorites/${id}`)
    } finally {
      loading.value = false
    }
  }
  
  async function addFavorite(data) {
    const result = await request('/favorites', {
      method: 'POST',
      body: JSON.stringify(data)
    })
    await loadFavorites()
    return result
  }
  
  async function updateFavorite(id, data) {
    await request(`/favorites/${id}`, {
      method: 'PUT',
      body: JSON.stringify(data)
    })
    await loadFavorites()
  }
  
  async function removeFavorite(id) {
    await request(`/favorites/${id}`, {
      method: 'DELETE'
    })
    favorites.value = favorites.value.filter(f => f.id !== id)
  }
  
  async function loadStats() {
    stats.value = await request('/favorites/stats/overview')
  }
  
  async function getRandomInspiration(count = 3) {
    return await request(`/favorites/random/inspiration?count=${count}`)
  }
  
  function setFilters(newFilters) {
    filters.value = { ...filters.value, ...newFilters }
    pagination.value.page = 1
    loadFavorites()
  }
  
  function resetFilters() {
    filters.value = {
      type: null,
      mood: null,
      search: ''
    }
    pagination.value.page = 1
    loadFavorites()
  }
  
  return {
    favorites,
    currentFavorite,
    stats,
    loading,
    filters,
    pagination,
    filteredFavorites,
    loadFavorites,
    loadFavorite,
    addFavorite,
    updateFavorite,
    removeFavorite,
    loadStats,
    getRandomInspiration,
    setFilters,
    resetFilters
  }
})
```

## 六、限制和注意事项

### 6.1 数据库限制
- ❌ **禁止删除或修改现有表**（users, diaries, pages, user_settings）
- ✅ 只能新增 favorites 表
- ✅ 使用外键关联，但设置为 SET NULL 避免级联删除问题
- ✅ 所有查询必须包含 user_id 过滤，确保数据隔离

### 6.2 前端限制
- ❌ **禁止修改现有页面的核心逻辑**（BookshelfView, DiaryReaderView, ArchiveView）
- ✅ 只重构 FavoritesView
- ✅ 保持现有的设计风格和动画效果
- ✅ 所有新组件必须遵循现有的 Tailwind CSS 类名规范
- ✅ 保持响应式设计（移动端 + 桌面端）

### 6.3 API 限制
- ❌ **禁止修改现有 API 端点**
- ✅ 只新增 /api/v1/favorites 相关端点
- ✅ 所有端点必须使用 authMiddleware 进行认证
- ✅ 遵循现有的响应格式：{ code, msg, data }

### 6.4 性能限制
- ✅ 单次查询最多返回 50 条记录
- ✅ 搜索结果最多返回 50 条
- ✅ 图片使用懒加载
- ✅ 大列表使用虚拟滚动（如果超过 100 条）

### 6.5 安全限制
- ✅ 所有输入必须进行 XSS 过滤
- ✅ 标签数量限制为最多 5 个
- ✅ 内容长度限制：
  - title: 最多 100 字符
  - content: 最多 1000 字符
  - source: 最多 100 字符
